#!/bin/sh
yum install git -y
git clone https://github.com/tomyankuns/Bootstrap.git /Bootstrap
yum install httpd -y
yum install tomcat7 -y
cp /Bootstrap/index.html /var/www/html/.
cp /Bootstrap/httpd.conf /etc/httpd/conf/.
cp /Bootstrap/httpd-proxy.conf /etc/httpd/conf/.
cp /Bootstrap/PicturePostService.war /usr/share/tomcat7/webapps/.
/etc/init.d/tomcat7 start
/etc/init.d/httpd start
